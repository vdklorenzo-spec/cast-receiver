package com.example.analogtowifispeakers import java.io.BufferedOutputStream import java.io.BufferedReader import java.io.OutputStream import java.io.InputStreamReader import java.net.ServerSocket import java.net.Socket import java.util.concurrent.CopyOnWriteArrayList import java.util.concurrent.atomic.AtomicBoolean import kotlin.concurrent.thread class LocalHttpServer( private val port: Int = 9090 ) { private data class Client( val socket: Socket, val out: OutputStream ) private var serverSocket: ServerSocket? = null private val running = AtomicBoolean(false) private val clients = CopyOnWriteArrayList() fun start() { if (running.get()) return running.set(true) serverSocket = ServerSocket(port) thread(name = "LocalHttpServer-Accept", isDaemon = true) { while (running.get()) { try { val client = serverSocket?.accept() ?: break handleClientAsync(client) } catch (_: Throwable) { // ignore accept errors during stop() } } } } fun stop() { running.set(false) try { serverSocket?.close() } catch (_: Throwable) {} serverSocket = null for (c in clients) { try { c.out.flush() } catch (_: Throwable) {} try { c.socket.close() } catch (_: Throwable) {} } clients.clear() } fun isRunning(): Boolean = running.get() fun pushAacFrame(frame: ByteArray) { if (!running.get()) return if (frame.isEmpty()) return if (clients.isEmpty()) return val chunkHeader = (frame.size.toString(16) + "\r\n").toByteArray(Charsets.US_ASCII) val chunkFooter = "\r\n".toByteArray(Charsets.US_ASCII) val toRemove = ArrayList() for (c in clients) { try { c.out.write(chunkHeader) c.out.write(frame) c.out.write(chunkFooter) c.out.flush() } catch (_: Throwable) { toRemove.add(c) } } if (toRemove.isNotEmpty()) { for (c in toRemove) { clients.remove(c) try { c.socket.close() } catch (_: Throwable) {} } } } private fun handleClientAsync(socket: Socket) { thread(name = "LocalHttpServer-Client", isDaemon = true) { socket.use { s -> try { s.soTimeout = 0 s.tcpNoDelay = true val reader = BufferedReader(InputStreamReader(s.getInputStream())) val requestLine = reader.readLine() ?: return@thread while (true) { val line = reader.readLine() ?: break if (line.isEmpty()) break } val parts = requestLine.split(" ") val rawPath = parts.getOrNull(1) ?: "/" val path = rawPath.substringBefore("?") val out = BufferedOutputStream(s.getOutputStream()) when { path == "/ping" -> respondPing(out) path == "/live.aac" -> registerLiveAacClient(s, out) path == "/receiver.html" -> respondReceiver(out) else -> respondNotFound(out) } } catch (_: Throwable) { // client closed / etc. } } } } private fun respondPing(out: BufferedOutputStream) { val body = "OK" val resp = buildString { append("HTTP/1.1 200 OK\r\n") append("Content-Type: text/plain\r\n") append("Content-Length: ${body.toByteArray(Charsets.UTF_8).size}\r\n") append("Connection: close\r\n") append("\r\n") append(body) } out.write(resp.toByteArray(Charsets.UTF_8)) out.flush() } private fun respondNotFound(out: BufferedOutputStream) { val body = "Not Found" val resp = buildString { append("HTTP/1.1 404 Not Found\r\n") append("Content-Type: text/plain\r\n") append("Content-Length: ${body.toByteArray(Charsets.UTF_8).size}\r\n") append("Connection: close\r\n") append("\r\n") append(body) } out.write(resp.toByteArray(Charsets.UTF_8)) out.flush() } private fun respondReceiver(out: BufferedOutputStream) { val html = """ AnalogToWifiSpeakers Receiver

AnalogToWifiSpeakers Receiver

Receiver booting...

""".trimIndent() val htmlBytes = html.toByteArray(Charsets.UTF_8) val resp = buildString { append("HTTP/1.1 200 OK\r\n") append("Content-Type: text/html; charset=utf-8\r\n") append("Content-Length: ${htmlBytes.size}\r\n") append("Cache-Control: no-cache, no-store, must-revalidate\r\n") append("Pragma: no-cache\r\n") append("Expires: 0\r\n") append("Connection: close\r\n") append("\r\n") } out.write(resp.toByteArray(Charsets.UTF_8)) out.write(htmlBytes) out.flush() } private fun registerLiveAacClient(socket: Socket, out: BufferedOutputStream) { val headers = buildString { append("HTTP/1.1 200 OK\r\n") append("Content-Type: audio/aac\r\n") append("Transfer-Encoding: chunked\r\n") append("Connection: keep-alive\r\n") append("Cache-Control: no-cache, no-store, must-revalidate\r\n") append("Pragma: no-cache\r\n") append("Expires: 0\r\n") append("X-Content-Type-Options: nosniff\r\n") append("\r\n") } out.write(headers.toByteArray(Charsets.UTF_8)) out.flush() clients.add(Client(socket, out)) while (running.get() && !socket.isClosed) { try { Thread.sleep(250) } catch (_: Throwable) { break } } clients.removeIf { it.socket == socket } try { socket.close() } catch (_: Throwable) {} } }
